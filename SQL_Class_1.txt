--- Command to create a TABLE
Create table if not exists employe
(
    id int,
    name VARCHAR(50),
    salary DOUBLE,
    hiring_date DATE
);

--- Syntax 1 To insert data into a TABLE
insert into employe values(1,'Shashank',1000,'2021-09-15');

--- This statement will fail
insert into employe values(2,'Rahul','2021-09-15');

--- Syntax 2 To insert data into a TABLE
insert into employe(salary,name,id) values(2000,'Rahul',2);

--- Insert multiple rows using single insert statement
insert into employe values(3,'Amit',3000,'2021-09-15'),(4,'Niting',3500,'2021-09-16'),(5,'Kajal',4000,'2021-09-20');

--- Use select command to query the data
Select * from employe;
